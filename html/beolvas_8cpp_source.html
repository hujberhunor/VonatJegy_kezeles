31Vonat beolvas(std::streampos& currPos){
32 const char* vonatok = "./input/vonatok.txt"; // const mert megadtam, hogy hol van és ehhez nem nyúlsz hozzá
33 std::ifstream file (vonatok); // Megynitom a file-t
34 Vonat von;
35
36 if (file.is_open() ){
37 // A currpos poziciora ugrom
38 file.seekg(currPos);
39 // Fájl beolvasása egy segéd struktúrába
40 Seged seged;
41 file >> seged.szam >> seged.indulo >> seged.veg >> seged.kocsidb >> seged.indulas >> seged.erkezes;
42
43 // Setterek, mert a konstruktor nem akarta megenni
44 von.setSzam(atoi(seged.szam));
45 von.setIndulo(seged.indulo);
46 von.setVeg(seged.veg);
47 von.setKocsidb(atoi(seged.kocsidb));
48 von.setIndulas(seged.indulas);
49 von.setErkezes(seged.erkezes);
50
51 // Minden új sor kezdeténél meg kell álnom és el kell tárolnom
52 // az éppenleges kurzor helyzetet, hogy a következő objektumba
53 // olvasás tudja hol kell folytatni.
54 char nextChar = file.peek();
55 if (nextChar == '\n') {
56 currPos = file.tellg(); // Kurzor helyzete.
57 } // end of peek
58 } // end of if

60 file.close();
